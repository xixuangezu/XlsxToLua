数据结构

	arr[TYPE]

		简易版tableString封装

		例: arr[int]		101;102;103
		
		输出: "name" = {[1] = 101, [2] = 102, [3] = 103}

	arr2[TYPE]

		二维数组

		例: arr2[int]		101,102,103;201,202;301,302,303
		
		输出: "name" = {[1] = {101,102,103}, [2] = {201,202}, [3] = {301,302,303}}

	tab{KEY=TYPE;KEY=TYPE;KEY={KEY=TYPE}}

		二维数组

		例: tab{key1=int;key2=int;key3={key4=int,key5=int}}		101;102;104,105
		
		输出: "name" = {key1 = 101; key2 = 102; key3 = {key4=104,key5=105}}
